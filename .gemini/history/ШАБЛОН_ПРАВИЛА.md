# 📝 Шаблон для заполнения компактных правил

## Как заполнять

Для каждого важного правила/термина создай компактную карточку.

---

## Шаблон SQL

```sql
INSERT INTO public.dgt_rules_compact (
  keyword, keyword_es, keyword_ru, topic_number,
  rule_summary,
  details,
  signs,
  terms,
  exam_tips,
  common_mistakes,
  related_rules,
  practical_example
) VALUES (
  'КЛЮЧЕВОЕ_СЛОВО',           -- английское/латиница (для ID)
  'término en español',        -- испанский термин
  'термин по-русски',          -- русский термин
  1,                           -- номер темы (1-10)
  
  'Краткая суть правила одним предложением.',
  
  '{"штраф": "200€", "баллы": 4, "другие_детали": "значения"}', -- JSON с цифрами
  
  ARRAY['R-5', 'R-6'],        -- знаки (если есть)
  ARRAY['термин1', 'término2'], -- ключевые термины
  
  ARRAY[                       -- советы для экзамена
    'Ловушка 1: что путают',
    'Ловушка 2: на что обратить внимание'
  ],
  
  ARRAY[                       -- частые ошибки
    'Путают с другим правилом',
    'Забывают про исключение'
  ],
  
  ARRAY['связанное_правило1', 'связанное_правило2'], -- ссылки на другие правила
  
  'Конкретный практический пример в 1-2 предложениях.'
);
```

## Примеры заполненных правил

### Пример 1: Estrechamiento (уже в базе)
```sql
INSERT INTO public.dgt_rules_compact VALUES (
  'estrechamiento',

  ARRAY['R-5', 'R-6'],
  ARRAY['estrechamiento', 'paso estrecho'],

```
