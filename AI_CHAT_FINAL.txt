// Search for compact rules (новый компактный формат)
async function searchCompactRules(query: string, topicNumber: number | null = null): Promise<string> {
  try {
    const supabase = createClient(
      Deno.env.get('SUPABASE_URL') ?? '',
      Deno.env.get('SUPABASE_ANON_KEY') ?? ''
    );

    console.log('[AI Chat] Searching compact rules for:', query.substring(0, 100));

    const { data, error } = await supabase
      .rpc('search_compact_rules', {
        search_query: query,
        topic_num: topicNumber,
        limit_count: 2 // Максимум 2 правила
      });

    if (error) {
      console.error('[AI Chat] Error searching compact rules:', error);
      return '';
    }

    if (!data || data.length === 0) {
      console.log('[AI Chat] No compact rules found');
      return '';
    }

    console.log(`[AI Chat] ✅ Found ${data.length} compact rules`);

    // Формируем компактный контекст
    const context = data
      .map((rule: any) => {
        let text = `**${rule.keyword_ru || rule.keyword}**: ${rule.rule_summary}`;
        
        if (rule.details) {
          const details = typeof rule.details === 'string' ? JSON.parse(rule.details) : rule.details;
          const detailsStr = Object.entries(details)
            .map(([k, v]) => `${k}: ${v}`)
            .join(', ');
          text += `\nДетали: ${detailsStr}`;
        }
        
        if (rule.signs && rule.signs.length > 0) {
          text += `\nЗнаки: ${rule.signs.join(', ')}`;
        }
        
        if (rule.exam_tips && rule.exam_tips.length > 0) {
          text += `\nЭкзамен: ${rule.exam_tips.join('; ')}`;
        }
        
        if (rule.practical_example) {
          text += `\nПример: ${rule.practical_example}`;
        }
        
        return text;
      })
      .join('\n\n---\n\n');

    return context;
  } catch (error) {
    console.error('[AI Chat] Exception searching compact rules:', error);
    return '';
  }
}

// Try Google Gemini API directly (free tier)
async function tryGemini(messages: any[], knowledge: string = ''): Promise<Response | null> {
  const GEMINI_API_KEY = Deno.env.get('GEMINI_API_KEY');
  
  if (!GEMINI_API_KEY) {
    console.log('[AI Chat] Gemini API key not found, skipping Gemini');
    return null;
  }

  try {
    console.log('[AI Chat] Trying Google Gemini API...');
    
    // Добавляем знания из базы в системный промпт
    let prompt = systemPrompt + '\n\n';
    
    if (knowledge) {
      prompt += `📚 МАТЕРИАЛ ИЗ УЧЕБНИКОВ DGT:\n${knowledge}\n\nИспользуй этот материал для точного ответа на основе официальных учебников DGT.\n\n`;
    }
    
    // Преобразуем формат сообщений для Gemini
    for (const msg of messages) {
      prompt += `${msg.role === 'user' ? 'User' : 'Assistant'}: ${msg.content}\n\n`;
    }
    
    const response = await fetch(
      `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-pro:streamGenerateContent?key=${GEMINI_API_KEY}`,
      {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          contents: [{
            parts: [{ text: prompt }]
          }],
          generationConfig: {
            temperature: 0.7,
            maxOutputTokens: 2000,
          }
        }),
      }
    );

    if (!response.ok) {
      const errorText = await response.text();
      console.error('[AI Chat] Gemini API error:', response.status, errorText);
      return null;
    }

    console.log('[AI Chat] ✅ Gemini API success');
    
    // Преобразуем ответ Gemini в формат SSE (Server-Sent Events)
    const reader = response.body?.getReader();
    if (!reader) {
      return null;
    }

    const stream = new ReadableStream({
      async start(controller) {
        const decoder = new TextDecoder();
        try {
          while (true) {
            const { done, value } = await reader.read();
            if (done) {
              controller.enqueue(new TextEncoder().encode('data: [DONE]\n\n'));
              controller.close();
              break;
            }

            const chunk = decoder.decode(value, { stream: true });
            const lines = chunk.split('\n').filter(line => line.trim());
            
            for (const line of lines) {
              try {
                const data = JSON.parse(line);
                const text = data.candidates?.[0]?.content?.parts?.[0]?.text;
                
                if (text) {
                  // Отправляем в формате OpenAI SSE
                  const sseData = {
                    choices: [{
                      delta: { content: text }
                    }]
                  };
                  controller.enqueue(new TextEncoder().encode(`data: ${JSON.stringify(sseData)}\n\n`));
                }
              } catch (e) {
                // Skip invalid JSON
              }
            }
          }
        } catch (error) {
          console.error('[AI Chat] Gemini streaming error:', error);
          controller.close();
        }
      }
    });

    return new Response(stream, {
      headers: { ...corsHeaders, 'Content-Type': 'text/event-stream' },
    });
  } catch (error) {
    console.error('[AI Chat] Gemini API exception:', error);
    return null;
  }
}

// Try Groq with vision support
async function tryGroq(messages: any[], knowledge: string = '', modelName: string = 'mixtral-8x7b-32768', imageUrl: string = ''): Promise<Response | null> {
  const GROQ_API_KEY = Deno.env.get('GROQ_API_KEY');
  
  if (!GROQ_API_KEY) {
    console.log('[AI Chat] Groq API key not found, skipping Groq');
    return null;
  }

  try {
    console.log(`[AI Chat] Trying Groq API with model: ${modelName}...`);
    
    // Добавляем знания из базы в системный промпт
    let enhancedPrompt = systemPrompt;
    
    if (knowledge) {
      enhancedPrompt += `

📚 СПРАВОЧНИК DGT (компактные правила):
${knowledge}

🎯 Используй эту информацию для точного ответа!
Расширь данные своими знаниями, но базируйся на этих фактах.`;
    }
    
    if (imageUrl) {
      enhancedPrompt += `

🖼️ ИЗОБРАЖЕНИЕ: ${imageUrl}
Проанализируй изображение и упоминай важные элементы (знаки, разметку, ситуацию на дороге), если они помогают понять правило. НЕ описывай очевидное.`;
    }

    const response = await fetch('https://api.groq.com/openai/v1/chat/completions', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${GROQ_API_KEY}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model: modelName,
        messages: [
          { role: 'system', content: enhancedPrompt },
          ...messages
        ],
        stream: true,
        temperature: 0.3, // Ещё ниже для минимума ошибок
        max_tokens: 3000, // Увеличено для развернутых ответов
        top_p: 0.9, // Более консервативная генерация
      }),
    });

    if (!response.ok) {
      const errorText = await response.text();
      console.error(`[AI Chat] Groq API error with ${modelName}:`, response.status, errorText);
      return null;
    }

    console.log(`[AI Chat] ✅ Groq API success with model: ${modelName}`);
    return new Response(response.body, {
      headers: { ...corsHeaders, 'Content-Type': 'text/event-stream' },
    });
  } catch (error) {
    console.error(`[AI Chat] Groq API exception with ${modelName}:`, error);
    return null;
  }
}

// Try multiple Groq models in order
async function tryGroqWithFallback(messages: any[], knowledge: string = '', imageUrl: string = ''): Promise<Response | null> {
  const models = [
    'llama-3.3-70b-versatile',   // Primary: best for Russian + vision
    // llama-3.1-8b убран - слишком маленький для промптов с учебниками
  ];

  for (const model of models) {
    const response = await tryGroq(messages, knowledge, model, imageUrl);
    if (response) {
      return response;
    }
  }

  return null;
}

Deno.serve(async (req) => {
  // Handle CORS preflight
  if (req.method === 'OPTIONS') {
    return new Response(null, { headers: corsHeaders });
  }

  try {
    console.log('[AI Chat] Request received:', {
      method: req.method,
      url: req.url,
      hasAuth: !!req.headers.get('Authorization')
    });

    // Parse request body
    let messages: any[] = [];
    let topicNumber: number | null = null;
    let imageUrl: string = '';
    try {
      const body = await req.json();
      messages = body.messages || [];
      topicNumber = body.topicNumber || null;
      imageUrl = body.imageUrl || '';
      
      if (topicNumber) {
        console.log(`[AI Chat] 🎯 Topic number provided: ${topicNumber} - will search in specific topic`);
      }
      
      if (imageUrl) {
        console.log(`[AI Chat] 🖼️ Image URL provided: ${imageUrl.substring(0, 50)}...`);
      }
    } catch (parseError) {
      console.error('[AI Chat] Failed to parse request body:', parseError);
      return new Response(
        JSON.stringify({ error: 'Invalid request body' }),
        { status: 400, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
      );
    }
    
    if (!Array.isArray(messages) || messages.length === 0) {
      return new Response(
        JSON.stringify({ error: 'Invalid messages format. Expected array of messages.' }),
        { status: 400, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
      );
    }

    // Get authorization header (optional)
    const authHeader = req.headers.get('Authorization');
    
    // Try to get user if auth header is provided (optional)
    if (authHeader) {
      try {
        const supabase = createClient(
          Deno.env.get('SUPABASE_URL') ?? '',
          Deno.env.get('SUPABASE_ANON_KEY') ?? '',
          {
            global: { headers: { Authorization: authHeader } }
          }
        );

        const { data: { user }, error: userError } = await supabase.auth.getUser();
        if (!userError && user) {
          console.log('[AI Chat] Authenticated user:', user.id);
        } else {
          console.log('[AI Chat] No authenticated user, continuing without auth');
        }
      } catch (authError) {
        console.log('[AI Chat] Auth check failed, continuing without auth');
      }
    } else {
      console.log('[AI Chat] No auth header provided, continuing without auth');
    }

    console.log('[AI Chat] Processing request with', messages.length, 'messages');

    // Search for relevant knowledge from DGT manuals
    const lastUserMessage = messages[messages.length - 1]?.content || '';
    
    // Extract key terms from the question for better search
    const extractKeyTerms = (text: string): string => {
      // Extract Spanish keywords (DGT terms)
      const spanishTerms = text.match(/\b(estrechamiento|prioridad|señal|autopista|autovía|rotonda|arcén|carril|adelantamiento|intersección|cruce|peatón|semáforo|paso|zona|velocidad|distancia|seguridad|multa|infracción)\w*/gi) || [];
      
      // Extract Russian keywords
      const russianTerms = text.match(/\b(сужени[ея]|приоритет|знак|обгон|перекр[её]сток|пешеход|светофор|скорость|расстояние|штраф)\w*/gi) || [];
      
      // Combine and deduplicate
      const allTerms = [...new Set([...spanishTerms, ...russianTerms])];
      
      // If no terms found, return first 100 chars
      return allTerms.length > 0 ? allTerms.join(' ') : text.substring(0, 100);
    };
    
    // Временно отключаем поиск учебников для экономии токенов
    // const searchQuery = extractKeyTerms(lastUserMessage);
    // console.log('[AI Chat] Search query extracted:', searchQuery);
    // const knowledge = await searchCompactRules(searchQuery, topicNumber);
    
    const knowledge = ''; // Используем только базовые знания AI
    console.log('[AI Chat] Using base AI knowledge (compact rules temporarily disabled)');

    // Use Groq as primary AI provider
    const groqResponse = await tryGroqWithFallback(messages, knowledge, imageUrl);
    if (groqResponse) {
      return groqResponse;
    }

    // No API available
    console.error('[AI Chat] ❌ No AI provider available');
    return new Response(
      JSON.stringify({ 
        error: 'AI service temporarily unavailable. Please configure GROQ_API_KEY in Supabase secrets.' 
      }),
      { 
        status: 503, 
        headers: { ...corsHeaders, 'Content-Type': 'application/json' } 
      }
    );
  } catch (e) {
    console.error('[AI Chat] Unexpected error:', e);
    return new Response(
      JSON.stringify({ error: e instanceof Error ? e.message : 'Unknown error' }), 
      {
        status: 500,
        headers: { ...corsHeaders, 'Content-Type': 'application/json' },
      }
    );
  }
});
