```typescript
// Строка 2463: Задержка 200ms
await new Promise(resolve => setTimeout(resolve, 200));

// Строки 2467-2485: Проверка статуса ПЕРЕД подсчетом
const { data: currentDuelStatus } = await supabase
  .from('duels')
  .select('status')
  .eq('id', duel_id)
  .single();

if (currentDuelStatus?.status === 'finished') {
  return { finished: true, reason: 'already_finished_by_opponent' };
}

// Строки 2571-2606: Атомарное обновление
const { data: updateResult } = await supabase
  .from('duels')
  .update({ status: 'finished', finished_at: new Date().toISOString() })
  .eq('id', duel_id)
  .eq('status', 'active') // ← КРИТИЧНО
  .select('id, status');

if (!updateResult || updateResult.length === 0) {
  return { finished: true, reason: 'already_finished_by_opponent_race' };
}
```
